# DIIS try to minimize the change of the input vectors. It rotates the vectors
# to minimize the error in the least square sense.
class DIIS(pyscf.lib.diis.DIIS):
    def __init__(self, mf, filename):
        pyscf.lib.diis.DIIS.__init__(self, mf, filename)
        self.rollback = False

# error vector = SDF-FDS
# error vector = F_ai ~ (S-SDS)*S^{-1}FDS = FDS - SDFDS ~ FDS-SDF in converge
    def update(self, s, d, f):
        if isinstance(f, numpy.ndarray) and f.ndim == 2:
            sdf = reduce(numpy.dot, (s,d,f))
            errvec = sdf.T.conj() - sdf
        else:
            sdf_a = reduce(numpy.dot, (s, d[0], f[0]))
            sdf_b = reduce(numpy.dot, (s, d[1], f[1]))
            errvec = numpy.hstack((sdf_a.T.conj() - sdf_a,
                                   sdf_b.T.conj() - sdf_b))
        logger.debug1(self, 'diis-norm(errvec)=%g', numpy.linalg.norm(errvec))
        xnew = pyscf.lib.diis.DIIS.update(self, f, xerr=errvec)
        if self.rollback > 0 and len(self._bookkeep) == self.space:
            self._bookkeep = self._bookkeep[-self.rollback:]
        return xnew

    def get_num_vec(self):
        if self.rollback:
            return self._head
        else:
            return len(self._bookkeep)

    def update(self, s, d, f):
        if not hasattr(self, '_mo_occ'):
            return f

        if self._mo_occ.ndim == 1 and 1. in self._mo_occ: # ROHF
            occidxa = mo_occ>0
            occidxb = mo_occ==2
            viridxa = mo_occ==0
            viridxb = mo_occ!=2
            nocca = occidxa.sum()
            noccb = occidxb.sum()
            nvira = viridxa.sum()
            nvirb = viridxb.sum()
            errvec = numpy.zeros((nocca,nvirb))
            errvec[:,:nvira] = \
                    reduce(numpy.dot, (self._mo_coeff[:,occidxa].T, f,
                                       self._mo_coeff[:,viridxa]))
            errvec[:noccb] += \
                    reduce(numpy.dot, (self._mo_coeff[:,occidxb].T, f,
                                       self._mo_coeff[:,viridxb]))
        elif self._mo_occ.ndim == 1: # RHF
            occidx = self._mo_occ > 0
            viridx = self._mo_occ == 0
            errvec = reduce(numpy.dot, (self._mo_coeff[:,occidx].T, f,
                                        self._mo_coeff[:,viridx]))
        else: # UHF
            occidxa = self._mo_occ[0]>0
            occidxb = self._mo_occ[1]>0
            viridxa = self._mo_occ[0]==0
            viridxb = self._mo_occ[1]==0
            errveca = reduce(numpy.dot, (self._mo_coeff[0][:,occidxa].T, f[0],
                                         self._mo_coeff[0][:,viridxa]))
            errvecb = reduce(numpy.dot, (self._mo_coeff[1][:,occidxb].T, f[1],
                                         self._mo_coeff[1][:,viridxb]))
            errvec = numpy.hstack((errveca.ravel(), errvecb.ravel()))
        logger.debug1(self, 'diis-norm(errvec)=%g', numpy.linalg.norm(errvec))
        return pyscf.lib.diis.DIIS.update(self, f, xerr=errvec)
